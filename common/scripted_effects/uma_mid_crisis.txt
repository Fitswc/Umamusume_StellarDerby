uma_mid_crisis_ship_creator_goldship = {
	create_leader = {
		class = commander
		species = owner_main_species
		name = random
		skill = 3
		randomize_traits = yes
		hide_age = yes
	}
	create_fleet = {
		effect = {
			set_owner = event_target:goldship_crisis_country
			create_ship = {
				name = random
				design = "NAME_uma_GOLDSHIP"
				graphical_culture = "extra_dimensional_01"
			}
			assign_leader = last_created_leader
			while = {
				count = 8
				create_ship = {
					name = random
					design = "NAME_uma_GOLDSHIP"
					graphical_culture = "extra_dimensional_01"
				}
			}
			while = {
				count = 12
				create_ship = {
					name = random
					design = "NAME_uma_GOLDSON"
					graphical_culture = "extra_dimensional_01"
				}
			}
			while = {
				count = 20
				create_ship = {
					name = random
					design = "NAME_uma_Wraith"
					graphical_culture = "extra_dimensional_01"
				}
			}
			set_location = {
				target = event_target:gold_ship_portal
				distance = 5
				angle = random
			}
			set_fleet_stance = aggressive
			set_aggro_range = 500
			set_aggro_range_measure_from = self
			set_fleet_bombardment_stance = armageddon

		}
	}
	last_created_fleet = {
		queue_actions = {
			repeat = {
				max_iterations = 2
				find_closest_system = {
					trigger = {
						id = uma_midcrisis.trigger.1
						owner = {
							has_authority = auth_trainingcenter
						}
					}
					found_system = {
						wait = {
							duration = 10
						}
						move_to = This
					}
				}
			}
		}
	}
}
