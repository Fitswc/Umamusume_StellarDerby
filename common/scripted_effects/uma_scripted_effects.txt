uma_school_planet_get_variable_effect = {
	export_trigger_value_to_variable = {
		trigger = num_districts
		parameters = {
			type = $DISTRICT$
		}
		variable = $VARIABLE$
	}
	divide_variable = { which = $VARIABLE$ value = 2 }
	floor_variable = $VARIABLE$
}

uma_upgrade_intelligence_goods_effect = {
	hidden_effect = {
		change_variable = {
			which = uma_intelligence_goods_used
			value = 1
		}
		if = {
			limit = { has_modifier = uma_intelligence_goods_basic_modifier }
			remove_modifier = uma_intelligence_goods_basic_modifier
		}
		if = {
			limit = { has_modifier = uma_intelligence_goods_special_modifier }
			remove_modifier = uma_intelligence_goods_special_modifier
		}
		add_modifier = {
			modifier = uma_intelligence_goods_basic_modifier
			multiplier = uma_intelligence_goods_used
			days = -1
		}
		if = {
			limit = {
				check_variable = {
					which = uma_intelligence_goods_basic_modifier
					value >= 10
				}
			}
			###ps:临时变量
			set_variable = {
				which = uma_intelligence_tenth_num
				value = uma_intelligence_goods_used
			}
			divide_variable = {
				which = uma_intelligence_tenth_num
				value = 10
			}
			floor_variable = uma_intelligence_tenth_num
			add_modifier = {
				modifier = uma_intelligence_goods_special_modifier
				multiplier = uma_intelligence_tenth_num
				days = -1
			}
			clear_variable = uma_intelligence_tenth_num
		}
	}
}
