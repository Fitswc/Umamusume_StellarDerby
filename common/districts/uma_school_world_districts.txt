district_uma_dormitory = {
	base_buildtime = 600
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

	show_on_uncolonized = {
		uses_district_set = uma_school_world
	}

	potential = {
		uses_district_set = uma_school_world
	}

	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
		district_arcology_housing
		district_arcology_organic_housing
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 500
			uma_crystal_carrot = 100
		}
		upkeep = {
			energy = 5
			uma_crystal_carrot = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 25
		job_clerk_add = 6
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_uma_country = yes }
		}
		job_uma_dormitory_administrator_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 9
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				has_ethic = ethic_gestalt_consciousness
			}
		}
		modifier = {
			job_maintenance_drone_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		modifier = {
			job_clerk_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = {
			job_clerk_add = 3
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_uma_country = yes }
		}
		text = job_uma_dormitory_administrator_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}

}
