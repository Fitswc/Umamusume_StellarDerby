building_uma_basic_racing_course = {
	base_buildtime = 100
	category = unity


	potential = {
		exists = owner
		owner = {
			OR = {
				has_authority = auth_trainingcenter
				#is_uma_fallen_empire = yes
			}
		}
		NOT = { has_modifier = slave_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				NOR = {
					has_authority = auth_trainingcenter
					#is_uma_fallen_empire = yes
				}
			}
			has_modifier = slave_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_authority = auth_trainingcenter
					#is_uma_fallen_empire = yes
				}
			}
		}
		modifier = {
			job_uma_racing_umamusume_add = 4
			job_uma_racing_ura_president_add = 1
			job_uma_racing_ura_members_add = 3
			job_uma_racing_ura_couse_worker_add = 4

		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				OR = {
					has_authority = auth_trainingcenter
					#is_uma_fallen_empire = yes
				}
			}
		}
		text = uma_job_uma_racing_umamusume_effect_desc
	}

	prerequisites = {
		"tech_uma_race_tradition"
	}

	upgrades = {
		building_uma_modern_racing_course
	}

	show_tech_unlock_if = {
		OR = {
			has_authority = auth_trainingcenter
			#is_uma_fallen_empire = yes
		}
	}

}
