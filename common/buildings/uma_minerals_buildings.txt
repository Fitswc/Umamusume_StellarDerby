	category = resource

		}

	destroy_trigger = {
		exists = owner

			owner = {

		}
	}

		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_authority = auth_trainingcenter
					#is_uma_fallen_empire = yes
				}
			}
		}
		modifier = {
			job_uma_core_miner_add = 2
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				OR = {
					has_authority = auth_trainingcenter
					#is_uma_fallen_empire = yes
				}
			}
		}
		text = job_uma_core_miner_effect_desc
	}

	prerequisites = {
		"tech_uma_core_mining"
	}

	upgrades = {
		building_uma_core_mining_complex_2
	}

	show_tech_unlock_if = {
		OR = {
			has_authority = auth_trainingcenter
			#is_uma_fallen_empire = yes
		}
	}

}
