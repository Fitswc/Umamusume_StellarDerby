namespace = uma_tradition

country_event = {
	id = uma_tradition.1
	hide_window = yes
#	is_triggered_only = yes
	mean_time_to_happen = {
		days = 10
	}
	trigger = {
		#has_traditions = uma_farm_self_cycle_final
		has_country_flag = uma_finish_tradition_self_cycle
	
	}
	immediate = {
		every_owned_planet = {
			limit = {
				num_pops >= 1
			}
			if = {
				limit = {
					OR = {
						num_assigned_jobs = {
							job = farmer
							value <= 15
						}
						num_assigned_jobs = {
							job = uma_carrot_farmer
							value <= 15
						}
						num_assigned_jobs = {
							job = uma_bee_farmer
							value <= 15
						}
					}
					NOT = {
						has_modifier = uma_tradition_FINAL
					}
				}
				add_modifier = {
					modifier = uma_tradition_FINAL
				}
			
			}
			else_if = {
				limit = {
					has_modifier = uma_tradition_FINAL
					num_assigned_jobs = {
						job = farmer
						value > 15
					}
				}
				remove_modifier = uma_tradition_FINAL
			}
		}
	}
}
