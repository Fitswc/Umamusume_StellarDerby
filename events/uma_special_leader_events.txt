country_event = {
	id = uma_special_leader.9000
	title = SKY
	desc = uma_special_leader.9000.desc
	is_triggered_only = yes

	picture = GFX_leader_recruitment_bg_legendary_azaryn
	event_window_type = leader_recruit
	picture_event_data = {
		portrait = event_target:uma_is_leader_seiun_sky #注意区分
	}

	immediate = {
		play_sound = uma_hire_Sky
		create_country = {
			name = "NAME_uma_empty"
			type = global_event
			auto_delete = no
			flag = {
				icon = {
					category = "special"
					file = "unknown.dds"
				}
				background = {
					category = "backgrounds"
					file = "00_solid.dds"
				}
				colors = {
					"red"
					"red"
					"null"
					"null"
				}
			}
			effect = {
				save_event_target_as = uma_empty_country
			}
		}
		event_target:uma_empty_country = {
			create_species = {
				name = "NAME_Unknown"
				class = umamusume
				portrait = umamusume
				namelist = "UMAname"
				traits = random
				effect = {
					save_event_target_as = uma_legendary_leader_species
				}
			}
			uma_create_leader_sky_admiral = yes
		}
	}

	option = {
		name = uma_special_leader.9000.a
		response_text = uma_special_leader.9000.a.resp
		is_dialog_only = yes
	}

	option = {
		name = uma_special_leader.9000.b
		response_text = uma_special_leader.9000.b.resp
		is_dialog_only = yes
	}

	option = {
		name = uma_special_leader.9000.c
		response_text = uma_special_leader.9000.c.resp
		is_dialog_only = yes
	}

	option = {
		name = HIRE
		tag = hire_leader
		custom_tooltip = uma_little_event.12.hire_tooltip #空的
		hidden_effect = {
			clone_leader = {
				target = event_target:uma_seiun_sky_admiral
				gender = female
				species = prev.species
			}
#			uma_create_leader_sky_admiral = yes
			last_created_leader = {
				save_global_event_target_as = uma_seiun_sky_admiral
				set_leader_flag = uma_is_special_leader
				set_leader_flag = uma_is_seiun_sky
				set_leader_flag = uma_is_ar_seiun_sky
				save_global_event_target_as = uma_is_leader_seiun_sky
				set_leader_flag = legendary_leader
				change_leader_portrait = "20c"
				#				add_trait = uma_ruler_trait_sp_seiun_sky #选举为领袖的时候才会保留特质！
				uma_initialize_leader_yaruki_effect = yes #随机添加心情特质
			}
			create_message = {
				type = MESSAGE_RECRUITED_LEADER
				localization = MESSAGE_RECRUITED_LEADER_DESC
				days = @toast_message_days
				target = event_target:uma_seiun_sky_admiral
				variable = {
					type = name
					localization = LEADER
					scope = event_target:uma_seiun_sky_admiral
				}
				variable = {
					key = "border"
					value = "GFX_border_legendary"
				}
			}
		}
	}

	option = {
		name = DISMISS
		tag = dismiss_leader
		hidden_effect = {
			event_target:uma_is_leader_seiun_sky = {
				kill_leader = { show_notification = no }
			}
		}
	}

	after = {
		hidden_effect = {
			event_target:uma_empty_country = {
				destroy_country = yes
			}
		}
	}
}
