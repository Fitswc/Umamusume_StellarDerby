	is_triggered_only = yes

	desc = uma_sweepy.1.desc

		NOT = {
			has_country_flag = uma_sweepy_evt_front_done
		}

		set_country_flag = uma_sweepy_evt_front_done
	}

	option = {
		name = uma_sweepy.1.a

			}
		}

		name = uma_sweepy.1.b
		hidden_effect = {
			add_resource = {
				influence = 20
			}
		}
	}
}
