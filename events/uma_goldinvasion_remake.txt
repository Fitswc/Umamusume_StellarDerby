country_event = {
	id = uma_crisis.520
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		has_global_flag = uma_gold_invasion_happened
		has_global_flag = uma_gold_ships_portal_still_here
	}

	immediate = {
		log = "Uma-GoldShip Invasion Happeneded -Part 1 Goldship Ship building cycle happened"
		event_target:goldship_crisis_country = {
			# 如果船的战力很强，打破，check later
			if = {
				limit = {
					has_global_flag = uma_global_game_setting_mid_game_crisis_diff_easy
					fleet_power > 10000
				}
				country_event = { id = uma_crisis.520 days = 200 }
				break = yes
			}
			if = {
				limit = {
					has_global_flag = uma_global_game_setting_mid_game_crisis_diff_normal
					fleet_power > 40000
				}
				country_event = { id = uma_crisis.520 days = 200 }
				break = yes
			}
			if = {
				limit = {
					has_global_flag = uma_global_game_setting_mid_game_crisis_diff_hard
					fleet_power > 70000
				}
				country_event = { id = uma_crisis.520 days = 200 }
				break = yes
			}

			if = {
				limit = {
					has_global_flag = uma_global_game_setting_mid_game_crisis_diff_easy
				}
				uma_mid_crisis_ship_creator_goldship = yes
				uma_mid_crisis_constructor_creator_gold = yes
			}
			if = {
				limit = {
					has_global_flag = uma_global_game_setting_mid_game_crisis_diff_normal
				}
				uma_mid_crisis_ship_creator_goldship = yes
				uma_mid_crisis_ship_creator_goldship = yes
				uma_mid_crisis_constructor_creator_gold = yes
			}
			if = {
				limit = {
					has_global_flag = uma_global_game_setting_mid_game_crisis_diff_hard
				}
				uma_mid_crisis_ship_creator_goldship = yes
				uma_mid_crisis_ship_creator_goldship = yes
				uma_mid_crisis_ship_creator_goldship = yes
				uma_mid_crisis_constructor_creator_gold = yes
			}
			country_event = { id = uma_crisis.520 days = 200 }
		}
	}
}
