						uma_is_pd_habitat_planet = yes

					}
				}
				distance = {
					source = root.solar_system
					min_jumps = 1
					max_jumps = 6
				}

					OR = {

					}
				}

					category = uma_soe_anolomy
				}

			}
		}
	}
}
