leader_event = {
	id = uma_traits_leader.11
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		owner = {
			has_authority = auth_trainingcenter
		}
		this = {
			has_leader_flag = uma_is_special_leader
		}
	}

	immediate = {
		if = {
			limit = {
				has_leader_flag = uma_is_tokai_teio
			}
			add_trait_no_notify = leader_trait_xian_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_oguri_cap
			}
			random_list = {
				50 = {
					add_trait_no_notify = leader_trait_xian_generic
				}
				50 = {
					add_trait_no_notify = leader_trait_cha_generic
				}
			}
		}

		if = {
			limit = {
				has_leader_flag = uma_is_agnes_digital
			}
			random_list = {
				50 = {
					add_trait_no_notify = leader_trait_xian_generic
				}
				50 = {
					add_trait_no_notify = leader_trait_cha_generic
				}
			}
		}

		if = {
			limit = {
				has_leader_flag = uma_is_seiun_sky
			}
			add_trait_no_notify = leader_trait_tao_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_tachyon
			}
			add_trait_no_notify = leader_trait_xian_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_silent_suzuka #AYB
			}
			add_trait_no_notify = leader_trait_zhui_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_nice_nature
			}
			add_trait_no_notify = leader_trait_cha_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_al_condor_pasa
			}
			random_list = {
				50 = {
					add_trait_no_notify = leader_trait_xian_generic
				}
				50 = {
					add_trait_no_notify = leader_trait_cha_generic
				}
			}
		}

		if = {
			limit = {
				has_leader_flag = uma_is_kitasan_black
			}
			add_trait_no_notify = leader_trait_tao_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_king_halo
			}
			add_trait_no_notify = leader_trait_cha_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_satono
			}
			add_trait_no_notify = leader_trait_cha_generic
		}

		if = {
			limit = {
				has_leader_flag = uma_is_deep
			}
			add_trait_no_notify = leader_trait_xian_generic
		}

		#笑
		if = {
			limit = {
				has_leader_flag = uma_is_glx
			}
			add_experience = 1
		}
	}
}

		FROMFROMFROM = {
			if = {
				limit = { exists = leader }

					if = {
						limit = { has_trait = leader_trait_datushuguan_scientists }

							add_resource = {

	}

	id = uma_traits_leader.5
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = leader
		leader = {
			has_trait_tier1or2or3 = {
				TRAIT = leader_trait_yingyan_scientists
			} 
		}
		FROM = {
			has_anomaly = no
			is_colonizable = no
			OR = {
				can_have_mineral_deposits = yes
				can_have_energy_deposits = yes
			}
			OR = {
				has_deposit_for = shipclass_mining_station
				num_deposits = 0
			}
			NOT = {
				has_planet_flag = prospected_by@root.owner
			}
		}
	}

	immediate = {
		if = {
			limit = {
				leader = {
					has_trait = leader_trait_yingyan_scientists
				}
			}
			scientist_prospector_effect = { CHANCE = 10 }
		}
		else = {
			scientist_prospector_effect = { CHANCE = 15 }
		}
	}
}

country_event = {
	id = uma_traits_leader.9
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		FROMFROM = { # Winning ship.
			if = {
				limit = { exists = fleet.leader }
				fleet.leader = {
					if = {
						limit = {
							OR = {
								has_trait = leader_trait_tianming1_admiral
							}
							ROOT.FROM = { # Owner of losing ship.
								is_default_or_fallen = yes
							}
						}
						if = {
							limit = { has_trait = leader_trait_tianming1_admiral }
							ROOT = { # Owner of winning ship.
								add_resource = {
									energy = 50
									alloys = 50
								}
							}
						}

					}
				}
			}
		}
	}				
}

planet_event = {
	id = uma_traits_leader.10
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_ground_combat = no }
			remove_planet_flag = checking_supreme_warrior
		}
		else = {
			# Handle both sides of the ground combat
			UMA_random_kill_army_if_supreme_warrior = { SIDE_A = attacker SIDE_B = defender }
			UMA_random_kill_army_if_supreme_warrior = { SIDE_A = defender SIDE_B = attacker }
	
			# Do the same check again in X days
			planet_event = { id = uma_traits_leader.10 days = 5 } # NOTE: Update custom loc for effect when changing days
		}
	}
}
